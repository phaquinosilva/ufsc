200 bool structures::Trie::Node::remove(const string& data_) {
201  if (data_.length() > 0) {
202  char letra = data_.at(0);
203  int filho_n = letra - char_a;
204  string resto_data;
205  resto_data.assign(data_.begin()+1, data_.end());
206 
207  if (filhos[filho_n] == nullptr) {
208  return false;
209  }
210  // possui a proxima letra
211  if (!resto_data.empty()) {
212  bool podeRemoverFilho = filhos[filho_n]->remove(resto_data);
213  // se for um prefixo de outra palavra nao posso remover
214  if (is_prefix() || filhos[filho_n]->possuiFilhos()) {
215  return false;
216  }
217  // vejo se posso remover o filho
218  if (podeRemoverFilho) {
219  // remove o filho e retorna q pode remover o proximo
220  delete filhos[filho_n];
221  filhos[filho_n] = nullptr;
222  }
223  return podeRemoverFilho;
224  }
225  // e a ultima letra
226  if (resto_data.empty() && filhos[filho_n]->is_prefix()) {
227  if (!filhos[filho_n]->possuiFilhos()) {
228  delete filhos[filho_n];
229  filhos[filho_n] = nullptr;
230  removidoAlgo = true;
231  return true;
232  }
233  return false;
234  }
235  }
236  return false;
237 }
