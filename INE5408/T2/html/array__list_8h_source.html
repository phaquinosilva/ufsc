106 template<typename T>
107 void structures::ArrayList<T>::insert(const T& data, std::size_t index) {
108  if (full()) {
109  throw std::out_of_range("lista cheia");
110  } else if (index > size_) {
111  throw std::out_of_range("fora da lista");
112  } else {
113  // percorro o array
114  for (std::size_t i = size_-1; i >= index && i < size_ ; i--) {
115  // copio o elemento pro proximo
116  contents[i+1] = contents[i];
117  }
118  contents[index] = data;
119  size_++;
120  }
121 }

125  if (full()) {

128  std::size_t i;

143 T structures::ArrayList<T>::pop(std::size_t index) {
144  if (empty()) {
145  throw std::out_of_range("lista vazia");
146  } else if (index > size_-1) {
147  throw std::out_of_range("fora da lista");
148  } else {
149  // salvo o item que ira ser removido
150  T item = contents[index];
151  // volto todos os elementos depois do item uma posicao
152  for (std::size_t i = index; i < size_-1; i++) {
153  contents[i] = contents[i+1];
154  }
155  size_--;
156  return item;
157  }
158  return 0;
